# Step 1: Use a base image that includes Python
FROM python:3.8-slim

# Step 2: Set the working directory inside the container
WORKDIR /app

# Step 3: Copy the requirements file and install dependencies (if you have one)
COPY requirements.txt .

# Step 4: Install the dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Step 5: Copy the current directory contents into the container at /app
COPY . .

# Step 6: Expose the port that the Flask app runs on
EXPOSE 5000

# Step 7: Define the command to run the app
CMD ["python", "app.py"]
